# Comandos para verificar usuarios en la base de datos

# Ver estructura de la tabla users
docker compose exec db psql -U hps_user -d hps_system -c "\d users"

# Ver todos los usuarios (sin role_name)
docker compose exec db psql -U hps_user -d hps_system -c "SELECT id, email, first_name, last_name, is_active FROM users;"

# Ver usuarios con su rol (si hay relaci√≥n con roles)
docker compose exec db psql -U hps_user -d hps_system -c "SELECT u.id, u.email, u.first_name, u.last_name, r.name as role_name, u.is_active FROM users u LEFT JOIN roles r ON u.role_id = r.id;"
